#### 练习1：基础操作

#### 练习5：复杂逻辑判断和后续操作

```python
# 答案5：复杂逻辑判断和后续操作
df['首次上市日期'] = pd.to_datetime(df['首次上市日期'])
filtered_df = df[df['首次上市日期'] < '1991-01-01']

def company_size(row):
    if row['注册资本（亿）'] > 100:
        return '大公司'
    else:
        return '小公司'

filtered_df['公司规模'] = filtered_df.apply(company_size, axis=1)
final_df = filtered_df

print(final_df)
```
